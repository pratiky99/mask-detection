def fibonacci(n):
    if(n<=1):
        print(n)
    else:
        fib1 = 0
        fib2 = 1
        print(fib1)
        for i in range(2,n+1):
            fn = fib1 + fib2
            fib2 = fib1
            fib1 = fn
            print(fn)

n = int(input("How many terms? "))
fibonacci(n)
